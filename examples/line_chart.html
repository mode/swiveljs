Line Chart Example

Total yards by season for each team in the NFL during the 2000 to 2013 seasons.

Sample Code

datasets.nfl(function(dataset) {
  var pivot = swivel
    .data(dataset.content)
    .fields('season', 'team')
    .pivots('team')
    .select(swivel.sum('yards'), 'Total Yards')
    .where(function(row) { return row['season'] >= 2000 && row['season'] < 2014; })
    .all();

  nv.addGraph(function() {
    var chart = nv.models.lineChart()
      .margin({left: 100})
      .showLegend(true)
      .showYAxis(true)
      .showXAxis(true);

    chart.xAxis
      .axisLabel('Season');

    chart.yAxis
      .axisLabel('Total Yards')
      .tickFormat(d3.format('.0f'));

    // Getting Data Together

    var data        = [];
    var axisName    = 'season';
    var seriesNames = pivot.values('team');

    // Initialize Series
    for(var s = 0; s < seriesNames.length; s++) {
      data.push({ key: seriesNames[s], values: [] });
    }

    // Populate Series
    for(var p = 0; p < pivot.data.length; p++) {
      var row = pivot.data[p];

      for(var s = 0; s < seriesNames.length; s++) {
        data[s].values.push({
          x: row[axisName],
          y: row[seriesNames[s]],
          label: row[axisName]
        });
      }
    }

    // End Getting Data Together

    d3.select('#chart svg').datum(data).call(chart);

    return chart;
  });
});
